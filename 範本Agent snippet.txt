                             tgSend("?? ?詨撌脤?雿銝?? + fullName + "???航?迂??啜??芸??詨霈?蝣箄???);
                             mainBody.click();
                         }
                     }, 500);
                 } else {
                     tgSend("⚠️ 找不到模型選單按鈕。");
                 }
             } else {
                 tgSend("⚠️ 不認識的代碼「" + targetArg + "」。輸入 /model 查看可用代碼。");
             }
         } else if (inputToSimulate.startsWith('/ask ') || inputToSimulate.startsWith('!')) {
             const cleanInput = inputToSimulate.startsWith('/ask ') ? inputToSimulate.substring(5) : inputToSimulate.substring(1);
             injectTextAndSend(cleanInput, state, true);
             tgSend("🗣️ 已幫您純粹作為文字輸入送出: " + cleanInput);
         } else {
             const remoteWrappedMessage = `請使用 <YOUR_SKILLS_PATH>\\00_remote-mode\\SKILL.md 技能處理以下請求：\n\n${inputToSimulate}`;
             injectTextAndSend(remoteWrappedMessage, state, true);
             tgSend("📩 已疊加遠端環境變數並為您發送: " + inputToSimulate);
         }
         didProcessSomething = true;
      } 
      
      if ((newPhotoMsg || newMsgs.length > 0) && !didProcessSomething) {
         tgSend("⚠️ 找不到編輯器的輸入框，請確認您的視窗已打開 AI 聊天室！");
      }

    } catch(e) {
       console.log("TG Poll Error", e);
    } finally {
      tgIsPolling = false;
    }
  };

  let chatObserver = null;
  let lastStreamTime = Date.now();
  
  const setupChatObserver = () => {
      if (chatObserver) {
          chatObserver.disconnect();
      }
      
      chatObserver = new MutationObserver(() => {
          if (!tgRunning) return;
          
          let currentlyThinking = false;
          const stopBtn = mainDoc.querySelector('div[data-tooltip-id="input-send-button-cancel-tooltip"]');
          if (stopBtn && (stopBtn.querySelector('.bg-red-500') || stopBtn.innerHTML.includes('bg-red-500'))) {
              currentlyThinking = true;
          }
          
          if (currentlyThinking) {
              const now = Date.now();
              if (now - lastStreamTime > 15000) {
                  lastStreamTime = now;
                  tgSend("✍️ AI 努力編輯中... (正在為您產生內容，請稍候)");
              }
          }
          
          if (wasBotThinking && !currentlyThinking) {
             console.log("[TG Monitor] Task finished. Extracting response payloads...");
             
             // 嘗試解析遠端廣播封包 
             let broadcastContent = "";
             let imageBox = "";
             try {
                 let fullText = mainBody.innerText || mainBody.textContent || "";
                 // 清除 VSCode 摺疊或奇怪的換行干擾
                 fullText = fullText.replace(/\u200B/g, '');

                 // 擷取最新的一筆文字總結 (找最後一個匹配)
                 const regexText = /\[TG_BROADCAST_START\]([\s\S]*?)\[TG_BROADCAST_END\]/g;
                 let matchText;
                 while ((matchText = regexText.exec(fullText)) !== null) {
                     broadcastContent = matchText[1].trim();
                 }
                 
                 // 擷取最新的一筆圖片 Base64
                 const regexImg = /\[TG_BASE64_IMAGE\]([\s\S]*?)\[TG_BASE64_IMAGE_END\]/g;
                 let matchImg;
                 while ((matchImg = regexImg.exec(fullText)) !== null) {
                     imageBox = matchImg[1].trim();
                 }
                 
             } catch(e) {
                 console.log("[TG Monitor] Error extracting broadcast:", e);
             }

             if (imageBox) {
                 tgSendPhoto(imageBox, `✅ 任務執行完畢！\n\n💬 AI 報告：\n${broadcastContent}`);
             } else if (broadcastContent) {
                 tgSend(`✅ 任務執行完畢！\n\n💬 AI 報告：\n${broadcastContent}`);
             } else {
                 tgSend("✅ 報告：Antigravity 任務已執行完畢！(AI 空閒中)\n*此任務未包含遠端摘要回報。*");
             }
          }
          wasBotThinking = currentlyThinking;
      });
      
      window.top.__tgObserver = chatObserver;

      chatObserver.observe(mainBody, {
          childList: true,
          subtree: true,
          attributes: true,
          attributeFilter: ['class', 'data-tooltip-id']
      });
  };

  tgBtn.onclick = (e) => {
    e.stopPropagation();
    if (!tgRunning) {
      if (window.top.__tgInterval) clearInterval(window.top.__tgInterval);
      
      tgInit = false; 
      window.top.__tgInterval = setInterval(pollTg, 4000);
      setupChatObserver();
      
      tgText.textContent = '⏸ 暫停 TG 遙控';
      tgBtn.style.background = greenGradient;
      tgRunning = true;
      tgSend("🚀 Antigravity 遠端遙控已連線... 隨時可以對我下達指令！");
    } else {
      clearInterval(window.top.__tgInterval);
      if (chatObserver) chatObserver.disconnect();
      
      tgText.textContent = '📱 開啟 TG 遙控';
      tgBtn.style.background = '#0088cc';
      tgRunning = false;
      tgSend("💤 遙控已斷線。");
    }
  };
  tgSection.appendChild(tgBtn);
  menu.appendChild(tgSection);

  const skillsTitle = mainDoc.createElement('div');
  skillsTitle.textContent = '⚡ Master Skills';
  skillsTitle.style.cssText = 'color: #aaa; font-size: 11px; margin-bottom: 6px; padding-left: 4px; border-top: none;';
  menu.appendChild(skillsTitle);

  masterSkills.forEach(skill => {
    const item = mainDoc.createElement('div');
    item.textContent = skill.replace('00_', '');
    item.title = '點擊複製: ' + skill;
    item.style.cssText = `
      display: block;
      width: 100%;
      min-height: 32px;
      padding: 8px 12px;
      margin-bottom: 2px;
      color: #e2e8f0;
      font-family: system-ui, -apple-system, sans-serif;
      font-size: 13px;
      line-height: normal; 
      cursor: pointer;
      border-radius: 4px;
      white-space: nowrap;
      overflow: hidden;
      text-overflow: ellipsis;
      box-sizing: border-box;
      user-select: none;
    `;
    item.onmouseover = () => {
      item.style.background = 'rgba(255, 255, 255, 0.1)';
      item.style.color = '#fff';
    };
    item.onmouseout = () => {
      item.style.background = 'transparent';
      item.style.color = '#e2e8f0';
    };
    item.onclick = (e) => {
      e.stopPropagation();
      const state = getChatState();
      const command = '請使用 <YOUR_SKILLS_PATH>\\' + skill;
      if (state.sendBox) {
         injectTextAndSend(command, state, false); // false = 不要自動發送
         const orig = item.textContent;
         item.textContent = '✅ 已為您輸入對話框！';
         item.style.color = '#4ade80';
         item.style.fontWeight = 'bold';
         setTimeout(() => {
           item.textContent = orig;
           item.style.color = '#e2e8f0';
           item.style.fontWeight = 'normal';
           menu.style.display = 'none';
           menuOpen = false;
         }, 700);
      } else {
         copyToClipboard(command);
         const orig = item.textContent;
         item.textContent = '✅ 已複製!';
         item.style.color = '#4ade80';
         item.style.fontWeight = 'bold';
         setTimeout(() => {
           item.textContent = orig;
           item.style.color = '#e2e8f0';
           item.style.fontWeight = 'normal';
           menu.style.display = 'none';
           menuOpen = false;
         }, 700);
      }
    };
    menu.appendChild(item);
  });

  btn.onclick = (e) => {
    e.stopPropagation();
    menuOpen = !menuOpen;
    menu.style.display = menuOpen ? 'block' : 'none';
  };
  
  btn.oncontextmenu = (e) => {
    e.preventDefault();
    if (window.top.__acInterval) clearInterval(window.top.__acInterval);
    btn.remove();
    menu.remove();
  };

  mainDoc.addEventListener('click', () => {
    if (menuOpen) {
      menu.style.display = 'none';
      menuOpen = false;
    }
  });

  mainBody.appendChild(btn);
  mainBody.appendChild(menu);
  console.log('🚀 7. Unified Agent Tools 完全載入成功！');
})();
