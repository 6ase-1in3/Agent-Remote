  const tgSendPhoto = async (base64Str, caption) => {

    try {

      // 1. Convert base64 to Blob

      const byteCharacters = atob(base64Str.replace(/\s/g, ''));

      const byteNumbers = new Array(byteCharacters.length);

      for (let i = 0; i < byteCharacters.length; i++) {

          byteNumbers[i] = byteCharacters.charCodeAt(i);

      }

      const byteArray = new Uint8Array(byteNumbers);

      const blob = new Blob([byteArray], {type: 'image/png'});



      // 2. Create FormData

      const formData = new FormData();

      formData.append('chat_id', TG_CHAT_ID);

      formData.append('photo', blob, 'screenshot.png');

      formData.append('caption', caption);



      // 3. Send

      await fetch('https://api.telegram.org/bot' + TG_TOKEN + '/sendPhoto', {

        method: 'POST',

        body: formData

      });

    } catch(e){

       console.log("TG Photo Send Error:", e);

  const injectTextAndSend = (textToSimulate, state, autoSend = true) => {

      if (!state.sendBox) return false;

      

      state.sendBox.focus();

      if (state.sendBox.tagName === 'TEXTAREA' || state.sendBox.tagName === 'INPUT') {

          const valueSetter = Object.getOwnPropertyDescriptor(state.sendBox, 'value')?.set;

          const prototype = Object.getPrototypeOf(state.sendBox);

          const prototypeValueSetter = Object.getOwnPropertyDescriptor(prototype, 'value')?.set;

          

          if (valueSetter && valueSetter !== prototypeValueSetter && prototypeValueSetter) {

             prototypeValueSetter.call(state.sendBox, textToSimulate);

          } else if (valueSetter) {

             valueSetter.call(state.sendBox, textToSimulate);

          } else {

             state.sendBox.value = textToSimulate;

          }

          state.sendBox.dispatchEvent(new Event('input', { bubbles: true }));

      } else {

          state.sendBox.focus();

          } catch(e) {}

          

          // Lexical 等富文本編輯器可能需要 Paste 方式

          const dataTransfer = new DataTransfer();

          dataTransfer.setData('text/plain', textToSimulate);

          const pasteEvent = new ClipboardEvent('paste', {

              bubbles: true,

              cancelable: true,

              clipboardData: dataTransfer

          });

          state.sendBox.dispatchEvent(pasteEvent);

          

          // 確保 Paste 失敗時，仍能插入文字
